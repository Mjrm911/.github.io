async function subscribe() { const permission = await Notification.requestPermission(); if (permission === 'granted') { const sw = await navigator.serviceWorker.ready; const sub = await sw.pushManager.subscribe({ userVisibleOnly: true, applicationServerKey: 'المفتاح_الخاص_بك' }); // الرابط اللي بيطلع لك الحين هو "جهاز التحكم" const token = btoa(JSON.stringify(sub)); prompt("انسخ هذا الكود وأرسله للمرسل (أو احفظه عندك):", token); alert("تم التفعيل! تقدر الحين تقفل الصفحة وتستقبل رسايل."); } }
